110 void mcp2515_configure_bit_timing(){
111  // Setting TQ duration: 0x3F in both mask and value gives the maximum prescaling
112  // and the lowest baudrate.
113  // TQ = 2*(BRP/F_OSC) = 63uS
114  mcp2515_BIT_MODIFY(MCP_CNF1, 0x3F, 0x3F);
115 
116  // Setting Synchronization Jump Width: 0x00 results in SJW = 1
117  mcp2515_BIT_MODIFY(MCP_CNF1, 0xC0, 0x00);
118 
119  // Setting PropSeg = 2 TQ
120  // CNF2 bits 2-0 (PRSEG2, 1, 0) configures the PropSeg length:
121  // (PRSEG + 1) x TQ. Thus PRSEG = 1 gives PropSeg = 2
122  mcp2515_BIT_MODIFY(MCP_CNF2, 0x07, 0x01);
123 
124  // Setting PS1 length:
125  // We target a length of 7 TQ.
126  // (PHSEG1 + 1) = 7 -> PHSEG1 = 6
127  // This is done with bit 5-3.
128  mcp2515_BIT_MODIFY(MCP_CNF2, 0x07 << 3, 0x06 << 3);
129 
130  // Setting PS2 length:
131  // We target a lenght of 6 TQ
132  // (PHSEG2 + 1) = 6 -> PHSEG2 = 5
133  mcp2515_BIT_MODIFY(MCP_CNF3, 0x07, 0x05);
134 }
