85 void mcp2515_configure_bit_timing(){
86  // Setting TQ duration: 0x3F in both mask and value gives the maximum prescaling
87  // and the lowest baudrate.
88  // TQ = 2*(BRP/F_OSC) = 63uS
89  mcp2515_BIT_MODIFY(MCP_CNF1, 0x3F, 0x3F);
90 
91  // Setting Synchronization Jump Width: 0x00 results in SJW = 1
92  mcp2515_BIT_MODIFY(MCP_CNF1, 0xC0, 0x00);
93 
94  // Setting PropSeg = 2 TQ
95  // CNF2 bits 2-0 (PRSEG2, 1, 0) configures the PropSeg length:
96  // (PRSEG + 1) x TQ. Thus PRSEG = 1 gives PropSeg = 2
97  mcp2515_BIT_MODIFY(MCP_CNF2, 0x07, 0x01);
98 
99  // Setting PS1 length:
100  // We target a length of 7 TQ.
101  // (PHSEG1 + 1) = 7 -> PHSEG1 = 6
102  // This is done with bit 5-3.
103  mcp2515_BIT_MODIFY(MCP_CNF2, 0x07 << 3, 0x06 << 3);
104 
105  // Setting PS2 length:
106  // We target a lenght of 6 TQ
107  // (PHSEG2 + 1) = 6 -> PHSEG2 = 5
108  mcp2515_BIT_MODIFY(MCP_CNF3, 0x07, 0x05);
109 }
